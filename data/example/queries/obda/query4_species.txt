
           
				PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
				PREFIX : <http://www.semanticweb.org/ontologies/2012/4/fishdelish.owl#>
				SELECT   ?genus ?Author ?family 
				WHERE { 
				$x a :Species .
				$x :genus  ?genus .
				$x :species  "multidens"^^xsd:string .
				$x :author ?Author .
				$x :belongsToFamily ?f .
				$f a :Family .
				$f :familyName  ?family .
				}
