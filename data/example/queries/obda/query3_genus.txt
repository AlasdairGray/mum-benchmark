
            
                PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
				PREFIX : <http://www.semanticweb.org/ontologies/2012/4/fishdelish.owl#>
				SELECT   ?species ?Author ?family
				WHERE { 
				$x a :Species .
				$x :genus  "Trichomycterus"^^xsd:string .
				$x :species  ?species .
				$x :author ?Author .
				$x :belongsToFamily ?y .
				$y a :Family .
				$y :familyName  ?family .
				}
