				PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
				SELECT ?genus ?species ?photographer ?stage  
                WHERE {
				$id a :Picture .
				$id :pictureName "Opsol_u1.jpg"^^xsd:string .
				$id :author ?photographer .
				$id :lifeStage ?stage .
				$id :isPictureOf ?x .
				$x a :Species .
				$x :genus ?genus .
				$x :species ?species .
                }
