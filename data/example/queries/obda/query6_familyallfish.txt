
            
                PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
				PREFIX : <http://www.semanticweb.org/ontologies/2012/4/fishdelish.owl#>
				SELECT ?Genus ?Species  ?author 
				WHERE { 
				$x a :Species .
				$x :genus ?Genus .
				$x :species ?Species .
				$x :author ?author .
				$x :belongsToFamily ?f .
				$f a :Family .
				$f :familyName "Triacanthodidae"^^xsd:string .
				}
