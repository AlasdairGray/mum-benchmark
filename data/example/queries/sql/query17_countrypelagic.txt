
            SELECT families.FamilyOrder , families.Family, species.Genus,species.Species,  country.Status, comnames.ComName, species.Dangerous
            FROM fbapp.families, fbapp.species, fbapp.comnames, fbapp.countref, fbapp.country
            WHERE countref.C_Code=country.C_Code
            AND comnames.SpecCode=species.SpecCode 
            AND country.SpecCode=species.SpecCode 
            AND species.FamCode = families.FamCode 
            AND countref.PAESE="St Helena"
            AND species.DemersPelag="pelagic"
